## 六、端到端 API 验证（curl）

```bash
# 清 Redis（避免脚本登录污染）
docker exec smart_energy_redis redis-cli FLUSHDB

# 测每个角色登录 + 看能访问的端点
for user in 2026010001 2026020001 2026030001 2026040001 2026050001 2026060001; do
  echo "=== 登录 $user ==="
  TOKEN=$(curl -s -X POST http://localhost:8080/api/auth/login \
    -H "Content-Type: application/json" \
    -d "{\"username\":\"$user\",\"password\":\"123456\"}" \
    | python -c "import json,sys; print(json.load(sys.stdin)['token'])" 2>/dev/null)
  if [ -z "$TOKEN" ]; then
    echo "  ❌ 登录失败"
    continue
  fi
  echo "  ✅ 登录成功"
  # 测 3 个端点
  for api in /api/work-orders /api/users/with-personnel /api/workorder/dispatch-board; do
    STATUS=$(curl -s -o /dev/null -w "%{http_code}" "http://localhost:8080${api}" -H "Authorization: Bearer $TOKEN")
    echo "    ${api}: ${STATUS}"
  done
  # 登出（清 session）
  curl -s -X POST http://localhost:8080/api/auth/logout -H "Authorization: Bearer $TOKEN" >/dev/null
done
```
